using System;
using System.Linq;
using System.Collections.Generic;


namespace functionalProgamming
{
    class Program
    {
        static void Main()
        {
            var pets = new[]
            {
                new {Name = "Sharo", Age = 6},
                new {Name = "Rex", Age = 4},
                new {Name = "Strela", Age = 1},
                new {Name = "Bora", Age = 3},
            };

            //var sortedPets = pets.OrderBy(pet => pet.Age); Сортира

            //var sortedPets = pets.OrderBy(pet => pet.Age); // Сортира по низходящ и по възраст

            //var sortedPets = pets.OrderBy(pet => pet.Name); // Сортира по име

            //var sortedPets = pets.OrderBy(pet => pet.Name.Length);// Сортира по дължина на името

            var sortedPets = pets.OrderBy(pet => pet.Name.Length).ThenBy(pet => pet.Name); // ThenBy (И след това по име)

            foreach (var pet in sortedPets)
            {
                Console.WriteLine("{0} --> {1}", pet.Name, pet.Age);
            }

        }
    }
}
